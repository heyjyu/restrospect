- reducer는 action.type === 'actiontype'를 이용하기 보다는 아래처럼 쓰면 깔끔하다.
```
const reducers = {
  actionType(...
  ,
};

function defaultReducer(state) {
  return state;
}

export default function reducer(state = initialState, action) {
  return (reducers[action.type] || defaultReducer)(state, action);
}
```
